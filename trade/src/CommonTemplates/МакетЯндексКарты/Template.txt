    function calcRouter () {
        var data = document.getElementById("data").value;
        if (!data) {
            //console.log('data is empty');
            return;
        }
        var ArrayPoints = JSON.parse(data);

        var routeinfo = document.getElementById('routeinfo').value;
        routeinfo = '';
        var Route = new ymaps.route(ArrayPoints);
        Route.then(function (route) {
            myMap.geoObjects.add(route);

            var points = route.getWayPoints();
            points.options.set('preset', 'islands#blueStretchyIcon');
            for (var i=0; i < points.getLength() ; i++) {
                var point = points.get(i);
                if (i == 0 || i == points.getLength()-1) {
                    point.properties.set('iconContent', 'Старт');
                    point.options.set('preset', 'islands#redStretchyIcon');
                } else {
                    point.properties.set('iconContent', i + ' - ' + ArrayPoints[i]);
                }
            }

            for (var i=0; i < route.getPaths().toArray().length; i++) {
                var curPath = route.getPaths().get(i);
                var length = curPath.getLength();
                routeinfo= routeinfo+length+';';
            }
            document.getElementById('routeinfo').value = routeinfo;

            var distance = route.getLength();
            document.getElementById('distance').value = distance;
        });
    }  
   
 function calcMultiRouter () {
        var data = document.getElementById("data").value;
        if (!data) {
            //console.log('data is empty');
            return;
        }
        var ArrayPoint = JSON.parse(data);
        //ArrayPoint = ["658087, Алтайский край, г. Новоалтайск, ул. Молодежная 26 В","659660, Алтайский край, село Сосновка, Лесная ул., д. 1","658087, Алтайский край, г. Новоалтайск, ул. Молодежная 26 В"];
        var multiRoute = new ymaps.multiRouter.MultiRoute({
            referencePoints: ArrayPoint,
        }, {
            editorDrawOver: false,
            wayPointDraggable: true,
            viaPointDraggable: true,
            // Задаем собственное оформление линий мультимаршрута.
            routeStrokeColor: "000088",
            routeActiveStrokeColor: "ff0000",
            pinIconFillColor: "ff0000",
            boundsAutoApply: true,
            zoomMargin: 30
        });
        myMap.geoObjects.add(multiRoute);

        multiRoute.events.once('update', function () {
            var activeRoute = multiRoute.getActiveRoute();
            var distance = activeRoute.properties._data.distance.value;
            var duration = activeRoute.properties._data.duration.value;
            var distance_text = activeRoute.properties._data.distance.text;
            var duration_text = activeRoute.properties._data.duration.text;

            document.getElementById('duration').value = distance;
            document.getElementById('distance').value = distance;

            document.getElementById('routeinfo').value = distance_text + '; ' +duration_text;
        });

    }
