    var myMap;
    const mapData = {}
    ymaps.ready(init);

    function init() {
        myMap = new ymaps.Map("map", {
            //center: [53.348053, 83.779875],
            center: mCordsFilial,
            zoom: 10
        });

        initMap();

        myMap.behaviors.enable('scrollZoom');
        myMap.behaviors.enable('router');
        myMap.behaviors.disable('dblClickZoom');
        myMap.controls.remove('searchControl');

        var searchControl = new ymaps.control.SearchControl({
            options: {
                noSelect: true,
                suppressYandexSearch: true,
                noPlacemark: true,
                maxWidth: 500,
                fitMaxWidth: true
            }
        });
        searchControl.events.add('load', function (event) {
            if (!event.get('skip') && searchControl.getResultsCount()) {
                searchControl.showResult(0);
                var data = searchControl.getResult(0)._value.properties._data;
                var geometry = searchControl.getResult(0)._value.geometry;

                mCordsPoint = geometry._coordinates;
                mapData.Address = data.text;
                mapData.mCords = mCordsPoint;

                addMarker(mCordsPoint, mapData.Address);
                getZones(mapData.mCords);

                setPopup();
            }
        });
        myMap.controls.add(searchControl);
        searchControl.search(startAddress)

        myMap.events.add('click', function (e) {
            var mCords = e.get('coords');

            //clear_mapData()
            clickPoligonHandler(mCords)

            mapData.mCords = mCords;
        });
    }//init
    
    function initMap() {
        // Удаляем все геообъекты на карте
        myMap.geoObjects.removeAll();

        clear_mapData()
        
        ViewZones();

        if (mapData.Address) {

    //Функции работа с зонами
    function ViewZones() {
        var strokeColor = '000';
        //var opacity = 0.5;
        var strokeWidth = 2;

        Zones.forEach(function (elem) {
            var arCoordsZone = GetCoordsZone(elem.sZone);
            var Polygon = new ymaps.Polygon(arCoordsZone, { hintContent: elem.hintContent }, { fillColor: elem.fillColor, strokeColor: strokeColor, strokeWidth: strokeWidth, opacity: elem.opacity, zone: elem.zone });
            AddEventPolygon(Polygon);
            elem.Polygon = Polygon;
        });
    }

    function GetCoordsZone(sZone) {
        var arCoordinates = Array();

        sZone = sZone.replace(/,0/g, '');
        var arTemp = sZone.split(' ');
        arTemp.forEach(function (item, i, arr) {
            var arCoords = item.split(',');
            arCoords = arCoords.reverse();
            arCoordinates[i] = arCoords;
        });
        var arCoordXY = Array(arCoordinates, []);

        return arCoordXY;
    }

    function AddEventPolygon(Polygon) {
        myMap.geoObjects.add(Polygon);

        Polygon.events.add('click', function (e) {
            var mCordsPoint = e.get('coords');
            var info = Polygon.properties._data.hintContent;

            getZones(mCordsPoint)
        });
    }

    const getZones = (mCordsPoint) => {
        //clear_mapData()
        const arCurZone = Zones.filter(function (currentValue, index) {
            if (currentValue.Polygon.geometry.contains(mCordsPoint)) return true;
        });

        if (arCurZone[0]) {
            const Polygon = arCurZone[0];
            mapData.zone = Polygon.zone;
            mapData.price = Polygon.price;
            mapData.mCords = mCordsPoint;
            mapData.Info = Polygon.hintContent;
            mapData.mCurZones = arCurZone;

            clickPoligonHandler(mCordsPoint);
            return true
        }

        clear_mapData()

        return false
    }

    function calcMultiRouter() {
        var ArrayPoint = [...mapData.mPoints]
        var multiRoute = new ymaps.multiRouter.MultiRoute({
            referencePoints: ArrayPoint,
        }, {
                editorDrawOver: false,
                wayPointDraggable: true,
                viaPointDraggable: true,
                // Задаем собственное оформление линий мультимаршрута.
                routeStrokeColor: "000088",
                routeActiveStrokeColor: "ff0000",
                pinIconFillColor: "ff0000",
                boundsAutoApply: true,
                zoomMargin: 30
            })

        myMap.geoObjects.add(multiRoute);

        multiRoute.events.add('activeroutechange', function () {
            var activeRoute = multiRoute.getActiveRoute();
            var Distance = activeRoute.properties._data.distance.value.toFixed(2);
            mapData.Distance = Distance * 2;
            var Duration = activeRoute.properties._data.duration.value;
            mapData.Duration = Duration * 2;
            var distance_text = activeRoute.properties._data.distance.text;
            var duration_text = activeRoute.properties._data.duration.text;
            mapData.Info = distance_text + '; ' + duration_text + ' в одну сторону';

            setPopup();
        });

        multiRoute.events.once('update', function () {    
        let routeinfo = ''
        var routes = multiRoute.getRoutes();
        for (var i = 0, l = routes.getLength(); i < l; i++) {
            var route = routes.get(i);
            var path = route.getPaths();
            const length = path.getLength();
            routeinfo = routeinfo + length + ';';
        }   

        })        
    }

    function calcRouter() {
        var ArrayPoints = [...mapData.mPoints]

        var Route = new ymaps.route(ArrayPoints);
        Route.then(function (route) {
            myMap.geoObjects.add(route);

            const points = route.getWayPoints()
            const path = route.getPaths()
            for (var i = 0; i < path.toArray().length; i++) {
                const curPath = path.get(i);
                const curPoint = points.get(i+1);

                const length = curPath.getLength();
                const lengthInfo = (length/1000).toFixed(2)+" км."

                curPoint.properties.set('iconContent', lengthInfo);
                curPoint.options.set('preset', 'islands#redStretchyIcon');

                mapData.мDestantions.push(length)
            }

            mapData.Distance = route.getLength()
            mapData.Duration = route.getTime()

            const distance_text = route.getHumanLength()
            const duration_text = route.getHumanTime()            

            mapData.Info = distance_text + '; ' + duration_text + ' в одну сторону';            

            setPopup();
        });
    }    

    const clickPoligonHandler = (mCordsPoint) => {
        ymaps.geocode(mCordsPoint).then(function (res) {
            var firstGeoObject = res.geoObjects.get(0);
            var address = firstGeoObject.getAddressLine();
            mapData.Address = address;
            addMarker(mCordsPoint, mapData.Info + ' (' + mapData.Address + ')');
            setPopup();
        });
    }

    function addMarker(coordPosition, name) {
        mapData.mPoints.push(coordPosition)
        var myPlacemark = new ymaps.Placemark(coordPosition, {
            // Свойства
            // Текст метки
            iconContent: name
        }, {
                // Опции
                // Иконка метки будет растягиваться под ее контент
                preset: 'islands#blueStretchyIcon'
            });
        myMap.geoObjects.add(myPlacemark);
    }

    function setGoordsOnAddress(Address, callback) {
        var myGeocoder = ymaps.geocode(Address);
        myGeocoder.then(
            function (res) {
                var firstGeoObject = res.geoObjects.get(0);
                if (firstGeoObject != null) {
                    var mCoords = firstGeoObject.geometry.getCoordinates();
                    callback(mCoords)
                }
            }
        );
    }

    function addMarkerOnAdress(Address) {
        var myGeocoder = ymaps.geocode(Address);
        myGeocoder.then(
            function (res) {
                var firstGeoObject = res.geoObjects.get(0);
                if (firstGeoObject != null) {
                    var mCoords = firstGeoObject.geometry.getCoordinates();
                    addMarker(mCoords, Address)
                }
            }
        );
    }
    
    function addMarkerOnCoords(lat, lon, name) {
		coordPosition = [lat, lon];
        mapData.mPoints.push(coordPosition)
        var myPlacemark = new ymaps.Placemark(coordPosition, {
            // Свойства
            // Текст метки
            iconContent: name
        }, {
                // Опции
                // Иконка метки будет растягиваться под ее контент
                preset: 'islands#blueStretchyIcon'
            });
        myMap.geoObjects.add(myPlacemark);
    }
    

    function setPopup() {
        let innerHTML = '<b>Информация: </b>' + mapData.Info;
        const popup = document.getElementById('popup');
        
        popup.innerHTML = innerHTML;
        popup.style.display = 'block';
    }

    const clear_mapData = () => {
        mapData.mCords = [];
        mapData.Address = startAddress;
        mapData.zone = '';
        mapData.price = 0;
        mapData.Destantion = 0;
        mapData.Duration = 0;
        mapData.distanseCost = 0;        
        mapData.Info = '';
        mapData.mCurZones = [];
        mapData.mPoints = [];
        mapData.мDestantions = [];
    }
