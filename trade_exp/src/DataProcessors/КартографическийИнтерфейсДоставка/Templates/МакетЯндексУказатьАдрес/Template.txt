<script type="text/javascript">
    var MyMap, ParamFor1C_Address, ParamFor1C_Cords; 
	            
	ymaps.ready(init);
	            
    function init(){
        myMap = new ymaps.Map ("map", {
            center: #КоординатыФилиала#,
        	zoom: 10
    });
	
    myMap.behaviors.enable('scrollZoom');
    myMap.behaviors.disable('dblClickZoom');

    myMap.events.add('click', function (e) {
        ParamFor1C_Cords = e.get('coords');
               
        myMap.geoObjects.removeAll();
        myMap.geoObjects.add(new ymaps.Placemark(ParamFor1C_Cords, { iconContent: '' }, { preset: 'islands#blueStretchyIcon' })); 
                                               
        var myReverseGeocoder = ymaps.geocode([ParamFor1C_Cords[0], ParamFor1C_Cords[1]]);
		myReverseGeocoder.then(function (res) {
				ParamFor1C_Address = res.geoObjects.get(0).getAddressLine();
            	setTimeout(() => dispatchEvent(new MouseEvent('click')), 1);
			});
    	});	
    } 
</script>
